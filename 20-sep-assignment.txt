#include <stdio.h>
#include<stdlib.h>
int main()
{
   char str[1000], ;
   int i, fre = 0;
   printf("Enter a string: ");
   gets(str);
   printf("Enter a character to find the frequency: ");
   scanf("%c",&ch);
   for(i = 0; str[i] != '\0'; i++)
   {
       if(ch == str[i])
           fre++;
   }
   printf("Frequency of %c = %d", ch, fre);
   return 0;
}
